import heapq
import numpy as np

def find_similar_vectors(embeddings, k, window):
  # sliding window with min-heap
  results = []
  heap = []
  for i in range(len(embeddings)):
    for j in range(i+1, min(i+window, len(embeddings))):
      sim = cosine_sim(embeddings[i], embeddings[j])
      if len(heap) < k:
        heapq.heappush(heap, (sim, i, j))
      elif sim > heap[0][0]:
        heapq.heapreplace(heap, (sim, i, j))
  return sorted(heap, reverse=True)
